Including UpSet Plots

define UpSet query functions

motifLength <- function(row, length) {
  newData <- (row["Unit.length"] %in% length)
}

STRlength <- function(row, min, max) {
  newData <- (as.numeric(row["STR.length"]) >= min) & (as.numeric(row["STR.length"]) <= max)
}
